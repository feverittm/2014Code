008import edu.wpi.first.wpilibj.Timer;

015 */
016public class ShootThenDriveAutonomous extends CommandGroup {
017    Timer myTimer;
018    public ShootThenDriveAutonomous() {
019        myTimer = new Timer();
020        myTimer.reset();
021        myTimer.start();
022        
023        addSequential(new PrepShooter());
024        addSequential(new WaitForHotGoal(myTimer));
025        addSequential(new Shoot());
026        addSequential(new DriveToSetpoint(RobotMap.distanceToDriveDuringAutonomous));
027        // Add Commands here:
028        // e.g. addSequential(new Command1());
029        //      addSequential(new Command2());
030        // these will run in order.

032        // To run multiple commands at the same time,
033        // use addParallel()
034        // e.g. addParallel(new Command1());
035        //      addSequential(new Command2());
036        // Command1 and Command2 will run in parallel.
037        // A command group will require all of the subsystems that each member
038        // would require.
039        // e.g. if Command1 requires chassis, and Command2 requires arm,
040        // a CommandGroup containing them would require both the chassis and the
041        // arm.
042    }
043}
