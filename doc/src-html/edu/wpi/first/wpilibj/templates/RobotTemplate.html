007package edu.wpi.first.wpilibj.templates;
008
009import edu.wpi.first.wpilibj.DriverStation;
010import edu.wpi.first.wpilibj.IterativeRobot;
011import edu.wpi.first.wpilibj.command.Command;
012import edu.wpi.first.wpilibj.command.Scheduler;
013import edu.wpi.first.wpilibj.livewindow.LiveWindow;
014import edu.wpi.first.wpilibj.templates.commands.CommandBase;
015import edu.wpi.first.wpilibj.templates.commands.DriveToUltrasonicThenFIre;
016
017/**
018 * The VM is configured to automatically run this class, and to call the
019 * functions corresponding to each mode, as described in the IterativeRobot
020 * documentation. If you change the name of this class or the package after
021 * creating this project, you must also update the manifest file in the resource
022 * directory.
023 */
024public class RobotTemplate extends IterativeRobot {
025
026    Command autonomousCommand;
027
028    /**
029     * Utility function to set the color of the LED's on the robot
030     * to match the team's alliance color.
031     */
032        public void setLedColor() {
033        if (DriverStation.getInstance().getAlliance() == DriverStation.Alliance.kRed) {
034            CommandBase.myLED.redON();
035        } else {
036            CommandBase.myLED.blueOn();
037        }
038    }
039
040    /**
041     * This function is run when the robot is first started up and should be
042     * used for any initialization code.
043     */
044    public void robotInit() {
045        // instantiate the command used for the autonomous period
046        setLedColor();
047
048        autonomousCommand = new DriveToUltrasonicThenFIre();
049
050        // Initialize all subsystems
051        CommandBase.init();
052    }
053
054    /**
055     *  Initialize robot for autonomous period
056     */
057    public void autonomousInit() {
058        // schedule the autonomous command (example)
059        autonomousCommand.start();
060
061        setLedColor();
062    }
063
064    /**
065     * This function is called periodically during autonomous
066     */
067    public void autonomousPeriodic() {
068        Scheduler.getInstance().run();
069    }
070
071    /**
072     *  carry out functions that need to occur at the end of autonomous 
073     */
074    public void disabledInit() {
075        setLedColor();
076    }
077
078    /**
079     * Set up to begin telop/human controlled period.
080     */
081    public void teleopInit() {
082        // This makes sure that the autonomous stops running when
083        // teleop starts running. If you want the autonomous to
084        // continue until interrupted by another command, remove
085        // this line or comment it out.
086        autonomousCommand.cancel();
087
088        setLedColor();
089        CommandBase.subGatherer.retractGatherer();
090        CommandBase.subShooter.unLatch();
091    }
092
093    /**
094     * This function is called periodically during operator control
095     */
096    public void teleopPeriodic() {
097        Scheduler.getInstance().run();
098        SmartDashboard();
099    }
100
101    /**
102     * This function is called periodically during test mode
103     */
104    public void testPeriodic() {
105        LiveWindow.run();
106    }
107
108    /**
109     *  Set up the callback functions to update status for the
110     *  smartdashboard.
111     */
112    public void SmartDashboard() {
113        //  CommandBase.myAimer.SmartDashboard();
114        CommandBase.oi.SmartDashboard();
115        CommandBase.subCompressor.SmartDashboard();
116        CommandBase.subDriveTrain.SmartDashboard();
117        CommandBase.subGatherer.SmartDashboard();
118        CommandBase.subShooter.SmartDashboard();
119    }
120}
